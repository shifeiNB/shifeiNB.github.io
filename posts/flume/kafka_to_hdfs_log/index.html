package com.fish.flume;

import com.google.gson.JsonElement;
import com.google.gson.JsonObject;
import com.google.gson.JsonParser;

import org.apache.flume.Context;
import org.apache.flume.Event;
import org.apache.flume.interceptor.Interceptor;

import java.nio.charset.StandardCharsets;
import java.util.List;
import java.util.Map;

/**
 * 将event body 的字节数组json获取ts 时间戳添加到header 的map中去
 1.header是一个map
 2.body是一个字节数组
 */
public class TimestampInterceptor implements Interceptor {
    private JsonParser json;

    /**
     *
     */
    @Override
    public void initialize() {
        /**初始化全局唯一json解析器*/
        json = new JsonParser();
    }

    /**
     * @param event
     * @return
     */
    @Override
    public Event intercept(Event event) {
        /**1.获取flume body，
         * 2.将body的timestamp 添加到kafka source 的json的 k  v
         * 3.json解析器解析字符串为json
         * 4.*/
        //获取flume body
        byte[] body = event.getBody();
//        将字节数组转换为字符串并解析为json
        String line = new String(body, StandardCharsets.UTF_8);

        JsonElement parse = json.parse(line);
        /**将json元素转换为json对象*/
        JsonObject jsonObject = parse.getAsJsonObject();
        /**获取到map的key*/
        JsonElement ts = jsonObject.get("ts");
        /**获取lone类型的ts*/
        long timestamp = ts.getAsLong();

        /**1.放到header中去/
        Map<String, String> headers = event.getHeaders();
        /**2.将ts 加到header  的 map中  key  timestamp*/
        Map<String, String> headers = event.getHeaders();
        headers.put("timestamp", Long.toString(timestamp));
        //返回事件对象
        return event;
    }

    /**
     * @param list
     * @return
     */
    @Override
    public List<Event> intercept(List<Event> list) {
        for (Event event : list) {
            intercept(event);
        }
        return list;
    }

    public static class Builder implements Interceptor.Builder {
        /**
         * @return
         */
        /**
         * 加载类文件时，初始化拦截器对象
         */
        @Override
        public Interceptor build() {
            return new TimestampInterceptor();
        }

        /**
         * @param context
         */
        @Override
        public void configure(Context context) {

        }

    }

    /**
     *
     */
    @Override
    public void close() {

    }
}
