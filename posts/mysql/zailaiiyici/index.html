# 首先卸载残留或不完整的MySQL
sudo bash /opt/software/remove_mysql.sh
# 安装MySQL的依赖
sudo yum install -y \
/opt/software/01_mysql-community-common-5.7.16-1.el7.x86_64.rpm \
/opt/software/02_mysql-community-libs-5.7.16-1.el7.x86_64.rpm \
/opt/software/03_mysql-community-libs-compat-5.7.16-1.el7.x86_64.rpm \
/opt/software/04_mysql-community-client-5.7.16-1.el7.x86_64.rpm \
/opt/software/05_mysql-community-server-5.7.16-1.el7.x86_64.rpm

# 启动MySQL
sudo systemctl start mysqld

# 获取临时密码
sudo cat /var/log/mysqld.log | grep 'temporary password' | awk '{print $11}'

# 登录并执行脚本
mysql -uroot -p"$(sudo cat /var/log/mysqld.log | grep 'temporary password' | awk '{print $11}')" --connect-expired-password < /opt/software/init.sql
