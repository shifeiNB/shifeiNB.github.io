-- 分区表
    -- 会额外增加一个用于分区的虚拟列
    -- 分区方式:
        -- 静态分区 ：指定分区字段的值
        -- 动态分区 ：
            -- 默认不支持（不开启），可以在hive-site.xml文件中开启配置
            -- 不指定分区字段的值，但是需要在赋值，额外增加一个值
CREATE EXTERNAL TABLE t_part_table
(
    `id`       BIGINT
) COMMENT '活动信息表'
    PARTITIONED BY (`dt` STRING)
    LOCATION '/test/t_part_table';

insert into table t_part_table partition (dt='2020-06-14') values (1);
set hive.exec.dynamic.partition.mode=nonstrict;
insert into table t_part_table partition (dt) values (1, '2020-06-15');

CREATE EXTERNAL TABLE t_loc_table
(
    `id`       BIGINT
);

-- 建立一张表，数据为JSON格式
    -- Hive默认情况下无法识别JSON格式的数据文件
drop table if exists t_json_table;
CREATE TABLE t_json_table
(
    `id`       BIGINT,
    `name`       STRING,
    `age`       BIGINT
)
    ROW FORMAT SERDE 'org.apache.hive.hcatalog.data.JsonSerDe'
    LOCATION '/test/t_json_table';

-- 特殊数据类型
    -- MySQL -> varchar -> Java -> String
    -- Hive -> SQL -> MR -> Java
    -- array
    -- map
    -- struts
drop table if exists t_datatype_table;
CREATE TABLE t_datatype_table
(
    `id`       array<int>,
    `kv`       MAP<int, int>,
    `obj`      struct<name:string>

)  LOCATION '/test/t_datatype_table';

-- array
select
    array_contains(ids, 2)
from (
     select `array`(1,2,3,4) ids
) t1;
-- map v类型要统一
    -- 判断map类型的数据中是否包含某一个key
select
    array_contains(map_keys(m), 'age')
from (
     select `map`('name', 'zhangsan', 'age', '30') m
) t2;
-- struct
select
    obj.name
from (
     select named_struct('name', 'zhangsan', 'age', 30) obj
) t3
