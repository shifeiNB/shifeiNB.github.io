-- ODS层
    -- Operate Data Store
    -- 数据仓库的数据源（Data Source）
        -- 解耦合：MySQL(Table) <-- DataX(Maxwell) --> HDFS(File) --> DW (HDFS, Table)
    -- ODS层为了后续的统计分析进行数据准备，那么应该和业务数据和日志数据尽量保持一致
        -- 异构系统数据的融合
        -- 简单数据的加工
        -- 数据格式尽量一致
            -- 日志: JSON
            -- 全量表: TSV
            -- 增量表: JSON

    -- 压缩格式
        -- gzip   : Hadoop默认支持压缩格式， 压缩比， 非常高， 压缩和解压缩效率不高
             -- ODS层存储全部历史数据，会占用大量存储空间, 能省一点是一点
             -- ODS层不进行数据的统计分析
        -- lzo    : 压缩比，不高不低， 压缩和解压缩的效率，不高不低
        -- snappy : 压缩比，不高，但是压缩和解压缩的效率非常高

    -- 表名： 见名知意
        -- ods_user_full

-- 日志表
    -- 建表语句
        -- 表的类型
            -- 内部表 ： Hive内部管理的表
            -- 外部表 ： Hive和外部的软件联合使用的表
        -- 分区表
            -- 根据分区字段dt,把表的数据拆分多个数据文件
    -- 数据格式
        -- JSON
        -- 如果JSON数据格式形成了嵌套
            -- 最外层的JSON对象的属性作为表的列
            -- 内层的JSON对象需要采用特殊数据类型

DROP TABLE IF EXISTS ods_log_inc;
CREATE EXTERNAL TABLE ods_log_inc
(
    `common`   STRUCT<ar :STRING,ba :STRING,ch :STRING,is_new :STRING,md :STRING,mid :STRING,os :STRING,uid :STRING,vc
                      :STRING> COMMENT '公共信息',
    `page`     STRUCT<during_time :STRING,item :STRING,item_type :STRING,last_page_id :STRING,page_id
                      :STRING,source_type :STRING> COMMENT '页面信息',
    `actions`  ARRAY<STRUCT<action_id:STRING,item:STRING,item_type:STRING,ts:BIGINT>> COMMENT '动作信息',
    `displays` ARRAY<STRUCT<display_type :STRING,item :STRING,item_type :STRING,`order` :STRING,pos_id
                            :STRING>> COMMENT '曝光信息',
    `start`    STRUCT<entry :STRING,loading_time :BIGINT,open_ad_id :BIGINT,open_ad_ms :BIGINT,open_ad_skip_ms
                      :BIGINT> COMMENT '启动信息',
    `err`      STRUCT<error_code:BIGINT,msg:STRING> COMMENT '错误信息',
    `ts`       BIGINT  COMMENT '时间戳'
) COMMENT '活动信息表'
    PARTITIONED BY (`dt` STRING)
    ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.JsonSerDe'
    LOCATION '/warehouse/gmall/ods/ods_log_inc/';
